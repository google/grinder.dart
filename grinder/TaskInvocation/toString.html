Source 

String toString() {
  var args = positionals.toList();
  options.forEach((name, value) {
    if (value != null) {
      args.add('$name=$value');
    }
  });
  var argString = args.isEmpty ? '' : ':${args.join(',')}';
  return '[$name$argString]';
}
